my note :

The Davies-Bouldin score, used to evaluate clustering quality, is calculated as $DB = \frac{1}{n_{clusters}} \sum_{i=1}^{n_{clusters}} \max_{j \neq i} \left( \frac{S_i + S_j}{M_{i,j}} \right)$, where $n_{clusters}$ is the number of clusters, $S_i$ is the within-cluster scatter for cluster $i$, and $M_{i,j}$ is the distance between the centroids of clusters $i$ and $j$.

Inertia measures the sum of squared distances between each data point and its assigned cluster centroid, and Distortion, in this context, is calculated as $ \text{Distortion} = \sqrt{\frac{\text{Inertia}}{N}} = \sqrt{\frac{\sum_{n=1}^{N} \sum_{k=1}^{K} r_{n,k} ||x_n - \mu_k||^2}{N}}$, where $N$ is the number of data points, $K$ is the number of clusters, $r_{n,k}$ is a binary indicator of whether data point $n$ belongs to cluster $k$, $x_n$ is the $n$-th data point, and $\mu_k$ is the centroid of cluster $k$.

import math
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

from sklearn.cluster import KMeans
from sklearn.preprocessing import MinMaxScaler
from sklearn.metrics import davies_bouldin_score

# load the MNIST (downsampled) dataset
raw_data = pd.read_csv("mnist-0.1.csv")
X = raw_data.drop("Label", axis=1)  # features
y = raw_data["Label"]              # not used for clustering

# scale the features (range 0 to 1)
scaler = MinMaxScaler()
X_scaled = scaler.fit_transform(X)

# k-means: compute DaviesBouldin score & Distortion for k=2..10
k_values = range(2, 11)
db_scores = []
distortions = []
inertias = []

for k in k_values:
    kmeans = KMeans(
        n_clusters=k,
        init="k-means++",
        random_state=0,
        n_init=10,
        max_iter=300
    )
    kmeans.fit(X_scaled)

    # labels from clustering
    labels = kmeans.labels_

    # DaviesBouldin score (lower is better)
    db_score = davies_bouldin_score(X_scaled, labels)
    db_scores.append(db_score)

    # inertia (sum of squared distances to cluster centers)
    inertia = kmeans.inertia_
    inertias.append(inertia)

    # distortion: sqrt(inertia / N)
    distortion = math.sqrt(inertia / X_scaled.shape[0])
    distortions.append(distortion)

    print(f"k={k}: DB score={db_score:.4f}, Inertia={inertia:.2f}, Distortion={distortion:.4f}")

# plot the DB score vs. k and Distortion vs. k in two subplots
fig, ax = plt.subplots(1, 2, figsize=(12, 5))

# DaviesBouldin Score vs. k
ax[0].plot(k_values, db_scores, marker='o')
ax[0].set_title("DaviesBouldin Score vs. k")
ax[0].set_xlabel("Number of Clusters (k)")
ax[0].set_ylabel("DaviesBouldin Score")

# distortion vs. k
ax[1].plot(k_values, distortions, marker='o', color='orange')
ax[1].set_title("Distortion vs. k")
ax[1].set_xlabel("Number of Clusters (k)")
ax[1].set_ylabel("Distortion (sqrt(Inertia/N))")

plt.tight_layout()
plt.show()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

from sklearn.preprocessing import MinMaxScaler
from scipy.cluster.hierarchy import dendrogram, linkage
from scipy.spatial.distance import pdist

# load the data
raw_data = pd.read_csv("mnist-0.1.csv")
X = raw_data.drop("Label", axis=1)
y = raw_data["Label"]  # not used for clustering, only for reference if needed

# scale the features (range 0 to 1)
scaler = MinMaxScaler()
X_scaled = scaler.fit_transform(X)

# compute the pairwise distance matrix
dist_matrix = pdist(X_scaled, metric='euclidean')

# perform hierarchical clustering using 'complete' linkage
linkage_matrix = linkage(dist_matrix, method='complete')

# visualize the hierarchy via a dendrogram
plt.figure(figsize=(15, 7))
dendrogram(linkage_matrix)
plt.title("Agglomerative Clustering Dendrogram (Complete Linkage)")
plt.xlabel("Sample Index")
plt.ylabel("Distance")
plt.show()

MNIST has 784 features (one per pixel) and with PCA, I project this data into a smaller subspace while retaining the bulk of its variance. this reducing dimensionality can speed up clustering and reduce memory usage.and also because PCA often discards noisy components, potentially improving clustering quality. for this task I used PCA method, but I found out that also exist other methods like t-SNE and UMAP that are accessible from sklearn, openTSNE, umap-learn libraries.

import math
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

from sklearn.preprocessing import MinMaxScaler
from sklearn.decomposition import PCA
from sklearn.cluster import KMeans
from sklearn.metrics import davies_bouldin_score

# load the MNIST (downsampled) dataset
raw_data = pd.read_csv("mnist-0.1.csv")
X = raw_data.drop("Label", axis=1)
y = raw_data["Label"]  # not used for clustering, only for reference

# scale the features to [0, 1]
scaler = MinMaxScaler()
X_scaled = scaler.fit_transform(X)

# apply PCA for dimensionality reduction
# number of components: sqrt(784) = 28 : I used heuristic sqrt(#original_feature) to reduce dimensionality while retaining a significant portion of the variance.
n_components = int(np.sqrt(X_scaled.shape[1]))  # 28 if 784 features
pca = PCA(n_components=n_components, random_state=0)
X_reduced = pca.fit_transform(X_scaled)

print(f"Original shape: {X_scaled.shape}")
print(f"Reduced shape after PCA: {X_reduced.shape}")

# evaluate K-means on the reduced data
k_values = range(2, 11)
db_scores_pca = []
distortions_pca = []
inertias_pca = []

for k in k_values:
    kmeans = KMeans(
        n_clusters=k,
        init="k-means++",
        random_state=0,
        n_init=10,
        max_iter=300
    )
    kmeans.fit(X_reduced)

    labels = kmeans.labels_
    db_score = davies_bouldin_score(X_reduced, labels)
    inertia = kmeans.inertia_
    distortion = math.sqrt(inertia / X_reduced.shape[0])

    db_scores_pca.append(db_score)
    distortions_pca.append(distortion)
    inertias_pca.append(inertia)

    print(f"k={k}, DB score={db_score:.4f}, Inertia={inertia:.2f}, Distortion={distortion:.4f}")

# plot the DB scores and distortion vs. k for PCA-reduced data
fig, ax = plt.subplots(1, 2, figsize=(12, 5))

# DaviesBouldin Score
ax[0].plot(k_values, db_scores_pca, marker='o')
ax[0].set_title("DaviesBouldin Score vs. k (PCA-Reduced)")
ax[0].set_xlabel("Number of Clusters (k)")
ax[0].set_ylabel("DaviesBouldin Score")

# Distortion
ax[1].plot(k_values, distortions_pca, marker='o', color='orange')
ax[1].set_title("Distortion vs. k (PCA-Reduced)")
ax[1].set_xlabel("Number of Clusters (k)")
ax[1].set_ylabel("Distortion (sqrt(Inertia/N))")

plt.tight_layout()
plt.show()

compare the DB score and distortion before and after PCA.¶

By reducing the data from 784 to 28 dimensions, PCA removes redundant or noisy features and makes the clusters more compact, as shown by the consistently lower DaviesBouldin scores (2.29 vs. 2.72 at k=2) and lower distortion values (5.38 vs. 6.40 at k=2). In contrast, without PCA, the high-dimensional data yields higher DB scores and distortion, indicating less compact clusters. Overall, applying PCA appears to improve cluster separation and compactness for this MNIST subset, making K-means more effective in the reduced feature space.
